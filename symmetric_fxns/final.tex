\begin{exercise}
  Prove that, for all $0 \leq k \leq n$, we have
  \begin{align*}
    e_k(\underbrace{1,\dots, 1}_{\text{n entries}}) = 
    \begin{pmatrix}
      n \\
      k
    \end{pmatrix}
    \quad \text{ and } \quad
    h_k(\underbrace{1,\dots, 1}_{\text{n entries}}) = 
    \begin{pmatrix}
      n + k - 1 \\
      k
    \end{pmatrix}.  
  \end{align*}

\end{exercise}
